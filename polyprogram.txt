import pandas as pd
from rdkit import Chem
from rdkit.Chem import Descriptors

# Read CSV instead of Excel
df = pd.read_csv("molecules.csv")

def extract_features(smiles):
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        return pd.Series([None, None, None])
    
    hetero_atoms = sum(1 for atom in mol.GetAtoms() if atom.GetAtomicNum() not in [1, 6])
    smiles_length = len(smiles)
    polar_surface_area = Descriptors.TPSA(mol)

    return pd.Series([hetero_atoms, smiles_length, polar_surface_area])

df[['hetero_atoms', 'smiles_length', 'polar_surface_area']] = df['SMILES'].apply(extract_features)

df.to_excel("output_program1.xlsx", index=False)

print("Program 1 done!")

import pandas as pd
from rdkit import Chem
from rdkit.Chem import Lipinski, Crippen

df = pd.read_excel("input.xlsx")

def extract_features(smiles):
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        return pd.Series([None, None, None, None])

    donors = Lipinski.NumHDonors(mol)
    acceptors = Lipinski.NumHAcceptors(mol)
    logp = Crippen.MolLogP(mol)
    molar_refractivity = Crippen.MolMR(mol)

    return pd.Series([donors, acceptors, logp, molar_refractivity])

df[['donors', 'acceptors', 'logp', 'molar_refractivity']] = df['SMILES'].apply(extract_features)

df.to_excel("output_program2.xlsx", index=False)
print("Program 2 done!")

import pandas as pd
from rdkit import Chem
from rdkit.Chem import Descriptors
from transformers import AutoTokenizer, AutoModel
import torch
import os
from transformers import logging
logging.set_verbosity_error()
os.environ["HF_HUB_DISABLE_SYMLINKS_WARNING"] = "1"


# ✅ Use working SMILES-BERT model
tokenizer = AutoTokenizer.from_pretrained("seyonec/PubChem10M_SMILES_BPE_450k")
model = AutoModel.from_pretrained("seyonec/PubChem10M_SMILES_BPE_450k")

df = pd.read_csv("molecules.csv")


def get_embedding(smiles):
    inputs = tokenizer(smiles, return_tensors="pt", truncation=True)
    with torch.no_grad():
        outputs = model(**inputs)

    return outputs.last_hidden_state[:, 0, :].squeeze().numpy()


def extract_features(smiles):
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        return pd.Series([None, None, None, None])

    topological_index = Descriptors.BalabanJ(mol)
    valence_electrons = Descriptors.NumValenceElectrons(mol)

    rotatable_bonds = Descriptors.NumRotatableBonds(mol)
    stability_index = 1 / (rotatable_bonds + 1)

    embedding = get_embedding(smiles)
    embedding_mean = embedding.mean()

    return pd.Series([topological_index, valence_electrons, stability_index, embedding_mean])


df[['topological_index', 'valence_electrons', 'stability_index', 'bert_embedding']] = df['SMILES'].apply(
    extract_features)

df.to_excel("output_program3.xlsx", index=False)

print("Program 3 fixed and running!")
